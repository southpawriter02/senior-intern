# Design Specification: The Senior Intern v0.3.5d "Enhanced Status Bar"

## Executive Summary

This document provides a detailed implementation specification for v0.3.5d, which creates a comprehensive status bar showing workspace, file, editor, and connection status. Each segment is interactive with tooltips and click actions, providing quick access to common functionality directly from the bottom of the window.

### v0.3.5d Scope

- Create `StatusBarViewModel` with all segment properties
- Create `StatusBar.axaml` UserControl with 10 segments
- Add workspace segment (folder icon + name, click to open)
- Add file segment (active file, click to reveal in explorer)
- Add language segment (detected language, click to change)
- Add cursor segment (line/column/selection, click for Go To Line)
- Add encoding segment (UTF-8 etc, click to change)
- Add line ending segment (LF/CRLF, click to change)
- Add unsaved indicator (yellow dot + count)
- Add file watcher status (sync icon)
- Add model status (brain icon + name/loading)
- Add tooltips for all segments

### Key Deliverables

| Deliverable | Description |
|-------------|-------------|
| StatusBarViewModel | ViewModel with all segments |
| StatusBar.axaml | 10-segment status bar UI |
| Workspace segment | Folder icon + name |
| File segment | Active file name |
| Cursor segment | Ln/Col + selection |
| Status indicators | Unsaved, watcher, model |
| Tooltips | Rich tooltips on all segments |
| Click actions | Commands for each segment |

namespace SeniorIntern.Desktop.ViewModels;

using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;

public partial class StatusBarViewModel : ViewModelBase
{
    #region Workspace Segment

    [ObservableProperty]
    private string? _workspaceName;

    [ObservableProperty]
    private bool _hasWorkspace;

    public string WorkspaceTooltip => HasWorkspace
        ? $"Workspace: {WorkspaceName}\nClick to open folder"
        : "No workspace open\nClick to open a folder";

    #endregion

    #region File Segment

    [ObservableProperty]
    private string? _activeFileName;

    [ObservableProperty]
    private bool _hasActiveFile;

    public string FileTooltip => HasActiveFile
        ? $"Active file: {ActiveFileName}\nClick to reveal in explorer"
        : "No file open";

    #endregion

    #region Language Segment

    [ObservableProperty]
    private string? _language;

    [ObservableProperty]
    private string? _languageDisplayName;

    public string LanguageTooltip => $"Language: {LanguageDisplayName ?? "Plain Text"}\nClick to change";

    #endregion

    #region Cursor Segment

    [ObservableProperty]
    private int _line = 1;

    [ObservableProperty]
    private int _column = 1;

    [ObservableProperty]
    private int _selectionLength;

    public string CursorDisplay => SelectionLength > 0
        ? $"Ln {Line}, Col {Column} ({SelectionLength} selected)"
        : $"Ln {Line}, Col {Column}";

    public string CursorTooltip => "Click to go to line";

    #endregion

    #region Encoding Segment

    [ObservableProperty]
    private string _encoding = "UTF-8";

    public string EncodingTooltip => $"Encoding: {Encoding}\nClick to change";

    #endregion

    #region Line Ending Segment

    [ObservableProperty]
    private string _lineEnding = "LF";

    public string LineEndingTooltip => $"Line ending: {LineEnding switch
    {
        "LF" => "Unix (LF)",
        "CRLF" => "Windows (CRLF)",
        "CR" => "Classic Mac (CR)",
        _ => LineEnding
    }}\nClick to change";

    #endregion

    #region Unsaved Indicator

    [ObservableProperty]
    private bool _hasUnsavedChanges;

    [ObservableProperty]
    private int _unsavedFilesCount;

    #endregion

    #region File Watcher Status

    [ObservableProperty]
    private bool _isFileWatcherActive;

    public string SyncTooltip => IsFileWatcherActive
        ? "File watcher active"
        : "File watcher inactive";

    #endregion

    #region Model Status

    [ObservableProperty]
    private string? _modelName;

    [ObservableProperty]
    private bool _isModelLoaded;

    [ObservableProperty]
    private bool _isModelLoading;

    public string ModelTooltip => IsModelLoaded
        ? $"Model: {ModelName}"
        : IsModelLoading
            ? "Loading model..."
            : "No model loaded";

    #endregion

    #region Commands

    [RelayCommand]
    private void OpenWorkspace()
    {
        OpenWorkspaceRequested?.Invoke(this, EventArgs.Empty);
    }

    [RelayCommand]
    private void RevealActiveFile()
    {
        if (HasActiveFile)
            RevealFileRequested?.Invoke(this, EventArgs.Empty);
    }

    [RelayCommand]
    private void ChangeLanguage()
    {
        ChangeLanguageRequested?.Invoke(this, EventArgs.Empty);
    }

    [RelayCommand]
    private void GoToLine()
    {
        GoToLineRequested?.Invoke(this, EventArgs.Empty);
    }

    [RelayCommand]
    private void ChangeEncoding()
    {
        ChangeEncodingRequested?.Invoke(this, EventArgs.Empty);
    }

    [RelayCommand]
    private void ChangeLineEnding()
    {
        ChangeLineEndingRequested?.Invoke(this, EventArgs.Empty);
    }

    #endregion

    #region Events

    public event EventHandler? OpenWorkspaceRequested;
    public event EventHandler? RevealFileRequested;
    public event EventHandler? ChangeLanguageRequested;
    public event EventHandler? GoToLineRequested;
    public event EventHandler? ChangeEncodingRequested;
    public event EventHandler? ChangeLineEndingRequested;

    #endregion

    #region Update Methods

    public void UpdateFromEditor(EditorTabViewModel? tab)
    {
        if (tab == null)
        {
            HasActiveFile = false;
            ActiveFileName = null;
            Language = null;
            LanguageDisplayName = null;
            Line = 1;
            Column = 1;
            SelectionLength = 0;
            Encoding = "UTF-8";
            LineEnding = "LF";
            return;
        }

        HasActiveFile = true;
        ActiveFileName = tab.FileName;
        Language = tab.Language;
        LanguageDisplayName = GetLanguageDisplayName(tab.Language);
        Line = tab.CaretLine;
        Column = tab.CaretColumn;
        SelectionLength = tab.SelectionLength;
        Encoding = tab.Encoding;
        LineEnding = tab.LineEnding;

        OnPropertyChanged(nameof(CursorDisplay));
        OnPropertyChanged(nameof(LanguageTooltip));
        OnPropertyChanged(nameof(EncodingTooltip));
        OnPropertyChanged(nameof(LineEndingTooltip));
    }

    public void UpdateFromWorkspace(Workspace? workspace)
    {
        HasWorkspace = workspace != null;
        WorkspaceName = workspace?.DisplayName;

        OnPropertyChanged(nameof(WorkspaceTooltip));
    }

    public void UpdateUnsavedStatus(int unsavedCount)
    {
        UnsavedFilesCount = unsavedCount;
        HasUnsavedChanges = unsavedCount > 0;
    }

    public void UpdateModelStatus(string? modelName, bool isLoaded, bool isLoading)
    {
        ModelName = modelName;
        IsModelLoaded = isLoaded;
        IsModelLoading = isLoading;

        OnPropertyChanged(nameof(ModelTooltip));
    }

    public void UpdateFileWatcher(bool isActive)
    {
        IsFileWatcherActive = isActive;

        OnPropertyChanged(nameof(SyncTooltip));
    }

    #endregion

    #region Helper Methods

    private static string? GetLanguageDisplayName(string? language)
    {
        return language?.ToLowerInvariant() switch
        {
            "csharp" => "C#",
            "javascript" => "JavaScript",
            "typescript" => "TypeScript",
            "javascriptreact" => "JavaScript React",
            "typescriptreact" => "TypeScript React",
            "python" => "Python",
            "json" => "JSON",
            "xml" => "XML",
            "xaml" => "XAML",
            "html" => "HTML",
            "css" => "CSS",
            "markdown" => "Markdown",
            "yaml" => "YAML",
            "shellscript" => "Shell Script",
            "powershell" => "PowerShell",
            "sql" => "SQL",
            "rust" => "Rust",
            "go" => "Go",
            "java" => "Java",
            "cpp" => "C++",
            "c" => "C",
            null => null,
            _ => language
        };
    }

    #endregion
}
