# Design Specification: AIntern v0.5.5h "Status Bar Integration"

## Overview

**Version**: v0.5.5h
**Parent**: v0.5.5 Polish & Integration
**Focus**: Terminal status information in the application status bar

### Purpose

This sub-version integrates terminal status into the status bar:
1. Create `TerminalStatusBarViewModel` with session tracking
2. Display active shell name and current directory
3. Show running process indicator badge
4. Display terminal count when multiple terminals open
5. Provide quick actions (toggle, new terminal, settings)
6. Support path abbreviation with ~ for home directory
7. Integrate with MainWindow status bar

### Dependencies

**From v0.5.1 (Terminal Foundation)**:
- `ITerminalService` events
- `TerminalSession` model

**From v0.5.3c (Shell Profile Models)**:
- `ShellProfile` for shell name

**Location**: `src/AIntern.Desktop/ViewModels/TerminalStatusBarViewModel.cs`

```csharp
namespace AIntern.Desktop.ViewModels;

using System.Timers;
using Avalonia;
using Avalonia.Threading;
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using Microsoft.Extensions.Logging;
using AIntern.Core.Interfaces;
using AIntern.Core.Models.Terminal;

/// <summary>
/// ViewModel for the terminal status bar section.
/// Displays active terminal info and provides quick actions.
/// </summary>
public partial class TerminalStatusBarViewModel : ViewModelBase, IDisposable
{
    private readonly ILogger<TerminalStatusBarViewModel> _logger;
    private readonly ITerminalService _terminalService;
    private readonly Timer _notificationTimer;
    private bool _isDisposed;

    // ========== Observable Properties ==========

    /// <summary>
    /// Whether there is an active terminal session.
    /// </summary>
    [ObservableProperty]
    private bool _hasActiveTerminal;

    /// <summary>
    /// Name of the active shell (e.g., "bash", "zsh", "PowerShell").
    /// </summary>
    [ObservableProperty]
    private string _activeShellName = string.Empty;

    /// <summary>
    /// Full path to the current working directory.
    /// </summary>
    [ObservableProperty]
    private string _currentDirectory = string.Empty;

    /// <summary>
    /// Abbreviated display version of the current directory.
    /// </summary>
    [ObservableProperty]
    private string _currentDirectoryDisplay = string.Empty;

    /// <summary>
    /// Whether a process is currently running in the terminal.
    /// </summary>
    [ObservableProperty]
    private bool _isProcessRunning;

    /// <summary>
    /// Name of the running process (e.g., "node", "python", "npm").
    /// </summary>
    [ObservableProperty]
    private string _runningProcessName = string.Empty;

    /// <summary>
    /// Total number of open terminal sessions.
    /// </summary>
    [ObservableProperty]
    private int _terminalCount;

    /// <summary>
    /// Whether a notification is currently being shown.
    /// </summary>
    [ObservableProperty]
    private bool _hasNotification;

    /// <summary>
    /// The notification message to display.
    /// </summary>
    [ObservableProperty]
    private string _notificationMessage = string.Empty;

    /// <summary>
    /// Whether the terminal panel is currently visible.
    /// </summary>
    [ObservableProperty]
    private bool _isTerminalVisible;

    /// <summary>
    /// Tooltip text showing full directory path.
    /// </summary>
    public string DirectoryTooltip => CurrentDirectory;

    /// <summary>
    /// Status tooltip with additional info.
    /// </summary>
    public string StatusTooltip => HasActiveTerminal
        ? $"{ActiveShellName} - {CurrentDirectory}"
        : "No active terminal. Click to open.";

    // ========== Constructor ==========

    public TerminalStatusBarViewModel(
        ILogger<TerminalStatusBarViewModel> logger,
        ITerminalService terminalService)
    {
        _logger = logger;
        _terminalService = terminalService;

        // Setup notification auto-hide timer
        _notificationTimer = new Timer(3000);
        _notificationTimer.Elapsed += OnNotificationTimerElapsed;
        _notificationTimer.AutoReset = false;

        // Subscribe to terminal events
        _terminalService.ActiveSessionChanged += OnActiveSessionChanged;
        _terminalService.SessionCreated += OnSessionCreated;
        _terminalService.SessionClosed += OnSessionClosed;
        _terminalService.VisibilityChanged += OnVisibilityChanged;

        // Initialize from current state
        UpdateTerminalCount();
        UpdateFromSession(_terminalService.ActiveSession);
        IsTerminalVisible = _terminalService.IsVisible;
    }

    // ========== Event Handlers ==========

    private void OnActiveSessionChanged(object? sender, TerminalSession? session)
    {
        Dispatcher.UIThread.Post(() => UpdateFromSession(session));
    }

    private void OnSessionCreated(object? sender, TerminalSession session)
    {
        Dispatcher.UIThread.Post(() =>
        {
            UpdateTerminalCount();
            _logger.LogDebug("Terminal session created: {Id}", session.Id);
        });
    }

    private void OnSessionClosed(object? sender, TerminalSession session)
    {
        Dispatcher.UIThread.Post(() =>
        {
            UpdateTerminalCount();
            _logger.LogDebug("Terminal session closed: {Id}", session.Id);
        });
    }

    private void OnVisibilityChanged(object? sender, bool isVisible)
    {
        Dispatcher.UIThread.Post(() =>
        {
            IsTerminalVisible = isVisible;
        });
    }

    private void OnNotificationTimerElapsed(object? sender, ElapsedEventArgs e)
    {
        Dispatcher.UIThread.Post(() =>
        {
            HasNotification = false;
            NotificationMessage = string.Empty;
        });
    }

    // ========== Update Methods ==========

    private void UpdateTerminalCount()
    {
        TerminalCount = _terminalService.Sessions.Count;
    }

    private void UpdateFromSession(TerminalSession? session)
    {
        HasActiveTerminal = session != null;

        if (session == null)
        {
            ActiveShellName = string.Empty;
            CurrentDirectory = string.Empty;
            CurrentDirectoryDisplay = string.Empty;
            IsProcessRunning = false;
            RunningProcessName = string.Empty;
            OnPropertyChanged(nameof(DirectoryTooltip));
            OnPropertyChanged(nameof(StatusTooltip));
            return;
        }

        ActiveShellName = session.Profile?.Name ?? session.ShellType.ToString();
        CurrentDirectory = session.CurrentDirectory ?? string.Empty;
        CurrentDirectoryDisplay = ShortenPath(CurrentDirectory);
        IsProcessRunning = session.HasRunningProcess;
        RunningProcessName = session.RunningProcessName ?? string.Empty;

        OnPropertyChanged(nameof(DirectoryTooltip));
        OnPropertyChanged(nameof(StatusTooltip));

        _logger.LogDebug(
            "Status bar updated: Shell={Shell}, Dir={Dir}, Process={Process}",
            ActiveShellName, CurrentDirectoryDisplay, RunningProcessName);
    }

    /// <summary>
    /// Abbreviates a path for display in the status bar.
    /// </summary>
    private static string ShortenPath(string path)
    {
        if (string.IsNullOrEmpty(path))
            return string.Empty;

        // Replace home directory with ~
        var home = Environment.GetFolderPath(Environment.SpecialFolder.UserProfile);
        if (!string.IsNullOrEmpty(home) && path.StartsWith(home, StringComparison.OrdinalIgnoreCase))
        {
            path = "~" + path[home.Length..];
        }

        // Truncate if too long
        const int maxLength = 40;
        if (path.Length > maxLength)
        {
            var separator = Path.DirectorySeparatorChar;
            var parts = path.Split(separator);
            
            if (parts.Length > 3)
            {
                // Show first part, ..., and last two parts
                return $"{parts[0]}{separator}...{separator}{parts[^2]}{separator}{parts[^1]}";
            }
        }

        return path;
    }

    // ========== Notification ==========

    /// <summary>
    /// Shows a temporary notification message.
    /// </summary>
    private void ShowNotification(string message)
    {
        NotificationMessage = message;
        HasNotification = true;

        // Reset and start the auto-hide timer
        _notificationTimer.Stop();
        _notificationTimer.Start();

        _logger.LogDebug("Showing notification: {Message}", message);
    }

    // ========== Commands ==========

    /// <summary>
    /// Toggles the terminal panel visibility.
    /// </summary>
    [RelayCommand]
    private void ToggleTerminal()
    {
        _terminalService.ToggleVisibility();
        _logger.LogDebug("Terminal visibility toggled");
    }

    /// <summary>
    /// Creates a new terminal session.
    /// </summary>
    [RelayCommand]
    private async Task NewTerminalAsync()
    {
        await _terminalService.CreateSessionAsync();
        _logger.LogInformation("New terminal session created from status bar");
    }

    /// <summary>
    /// Opens the terminal settings panel.
    /// </summary>
    [RelayCommand]
    private void OpenTerminalSettings()
    {
        TerminalSettingsRequested?.Invoke(this, EventArgs.Empty);
    }

    /// <summary>
    /// Copies the current directory path to the clipboard.
    /// </summary>
    [RelayCommand]
    private async Task CopyCurrentDirectoryAsync()
    {
        if (string.IsNullOrEmpty(CurrentDirectory))
            return;

        var clipboard = Application.Current?.Clipboard;
        if (clipboard != null)
        {
            await clipboard.SetTextAsync(CurrentDirectory);
            ShowNotification("Path copied to clipboard");
        }
    }

    /// <summary>
    /// Opens the current directory in the file explorer.
    /// </summary>
    [RelayCommand]
    private void OpenInFileExplorer()
    {
        if (string.IsNullOrEmpty(CurrentDirectory) || !Directory.Exists(CurrentDirectory))
            return;

        try
        {
            System.Diagnostics.Process.Start(new System.Diagnostics.ProcessStartInfo
            {
                FileName = CurrentDirectory,
                UseShellExecute = true
            });
        }
        catch (Exception ex)
        {
            _logger.LogWarning(ex, "Failed to open directory in file explorer: {Path}", CurrentDirectory);
        }
    }

    /// <summary>
    /// Cycles to the next terminal session.
    /// </summary>
    [RelayCommand]
    private void NextTerminal()
    {
        _terminalService.ActivateNextSession();
    }

    /// <summary>
    /// Cycles to the previous terminal session.
    /// </summary>
    [RelayCommand]
    private void PreviousTerminal()
    {
        _terminalService.ActivatePreviousSession();
    }

    // ========== Events ==========

    /// <summary>
    /// Raised when terminal settings should be opened.
    /// </summary>
    public event EventHandler? TerminalSettingsRequested;

    // ========== IDisposable ==========

    public void Dispose()
    {
        if (_isDisposed)
            return;

        _terminalService.ActiveSessionChanged -= OnActiveSessionChanged;
        _terminalService.SessionCreated -= OnSessionCreated;
        _terminalService.SessionClosed -= OnSessionClosed;
        _terminalService.VisibilityChanged -= OnVisibilityChanged;

        _notificationTimer.Stop();
        _notificationTimer.Dispose();

        _isDisposed = true;
    }
}
```

```csharp
namespace AIntern.Desktop.Converters;

using System.Globalization;
using Avalonia.Data.Converters;

/// <summary>
/// Returns true if the value is greater than zero.
/// </summary>
public class GreaterThanZeroConverter : IValueConverter
{
    public object? Convert(object? value, Type targetType, object? parameter, CultureInfo culture)
    {
        if (value is int intValue)
            return intValue > 0;
        
        if (value is double doubleValue)
            return doubleValue > 0;

        return false;
    }

    public object? ConvertBack(object? value, Type targetType, object? parameter, CultureInfo culture)
    {
        throw new NotSupportedException();
    }
}

/// <summary>
/// Returns true if the value is greater than one.
/// </summary>
public class GreaterThanOneConverter : IValueConverter
{
    public object? Convert(object? value, Type targetType, object? parameter, CultureInfo culture)
    {
        if (value is int intValue)
            return intValue > 1;
        
        if (value is double doubleValue)
            return doubleValue > 1;

        return false;
    }

    public object? ConvertBack(object? value, Type targetType, object? parameter, CultureInfo culture)
    {
        throw new NotSupportedException();
    }
}
```
